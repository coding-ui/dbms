SELECT NAME, COUNT(*) AS name_count
FROM employees
GROUP BY NAME
HAVING COUNT(*) > 1;

SELECT *
FROM employees
WHERE NAME = 'Amit' AND SALARY > 50000;

SELECT BRANCH, COUNT(*) AS num_employees
FROM employees
GROUP BY BRANCH;

SELECT NAME
FROM employees
WHERE NAME LIKE 'S%';

SELECT NAME
FROM employees
WHERE DOJ BETWEEN TO_DATE('01-JAN-1990', 'DD-MON-YYYY') AND TO_DATE('31-DEC-1999', 'DD-MON-YYYY');
